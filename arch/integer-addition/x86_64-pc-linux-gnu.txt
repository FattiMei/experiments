add_int64_t:
	leaq	(%rdi,%rsi), %rax
	retq

add_uint64_t:
	leaq	(%rdi,%rsi), %rax
	retq

add_int128_t:
	movq	%rdi, %rax
	addq	%rdx, %rax
	adcq	%rcx, %rsi
	movq	%rsi, %rdx
	retq
